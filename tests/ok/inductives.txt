\module Ind {
  \inductive Unit: \Set :=
  | unit: Unit ;
  ;

  \inductive Bool: \Set :=
  | true: Bool ;
  | false: Bool ;
  ;

  \inductive Nat: \Set :=
  | zero: Nat ;
  | succ: Nat -> Nat ;
  ;

  \definition one: Nat := Nat.succ Nat.zero;
  \definition two: Nat := Nat.succ (Nat.succ Nat.zero);
}
